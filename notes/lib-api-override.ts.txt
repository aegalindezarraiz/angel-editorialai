export async function fetchAnalyticsData() {
  const response = await fetch('/api/analytics/mission-local-001', { cache: 'no-store' });
  const data = await response.json();
  return data.analytics;
}

export async function fetchPipelineData() {
  const response = await fetch('/api/pipeline/mission-local-001', { cache: 'no-store' });
  const data = await response.json();
  return data.stages ?? [];
}

export async function submitMission(input: Record<string, unknown>) {
  const response = await fetch('/api/missions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(input)
  });

  if (!response.ok) {
    throw new Error('Failed to submit mission');
  }

  return response.json();
}
